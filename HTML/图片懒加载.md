### 原理思路
- 拿到图片的img dom
- 重点第二步,判断当前图片是否到了可视区域范围内
- 到了可视区域的高度以后,就将img的data-src属性设置给src
- 绑定window的scroll事件
```css
img {
  display: block;
  height:320px;
  margin-top:20px;
  margin:10px auto;
}
```

```html
<img src="default.png" data-scr="https:chaojidaofou/user.png" />
```
```js
let Img = document.getElementbyTagName('img'),
    len = Img.length,
    count = 0;

function lazyLoad() {
  let viewH = document.body.clientHeight, // 可见区域宽度
      scrollTop = document.body.scrollTop; // 滚动条距离顶部高度
  for (let i = count;i <len; i++) {
    if (Img[i].getAttribute('src') === 'default.png') {
      Img[i].src = Img[i].getAttribute('data-src')
      count++;
    }
  }
}

// 节流
function throttle(fn,wait,immediate) {
  let timer = null
  let callNow = immediate
  return function () {
    let context = this;
      args = arguments
    if(callNow) {
      fn.apply(context,args)
      callNow = false
    }
    if(!timer) {
      timer = setTimeout(() => {
        fn.apply(context, args)
        timer = null
      },wait)
    }
  }
}
```
